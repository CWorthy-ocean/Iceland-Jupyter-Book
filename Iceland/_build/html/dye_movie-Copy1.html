from matplotlib import pyplot as plt
from celluloid import Camera
from IPython.display import HTML
fig = plt.figure()
levels=[0.1,10,0.1]
cmap1='Reds'
cmap2='Blues'
cmap3='Greens'
cmap4='Oranges'
k1=0
k2=1
k3=0.1
levels = np.arange(k1, k2, k3)
cmap = 'jet'
camera = Camera(fig)
for i in range(48):
    plt.contourf(dye1_plot.isel(time=i).where(dye1_plot.isel(time=i)!=0),levels, extend="both", cmap=cmap1)
    plt.contourf(dye2_plot.isel(time=i).where(dye2_plot.isel(time=i)!=0),levels, extend="both", cmap=cmap2)
    plt.contourf(dye3_plot.isel(time=i).where(dye3_plot.isel(time=i)!=0),levels, extend="both", cmap=cmap3)
 
    camera.snap()
    plt.contourf(mask_rho.where(mask_rho!=1),cmap='Greys')
    plt.title('2 dye lines over 48 hours')
animation = camera.animate()
HTML(animation.to_html5_video())

animation.save('textmovie.mp4')
